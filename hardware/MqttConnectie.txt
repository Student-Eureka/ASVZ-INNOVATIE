# main.py (MicroPython - Raspberry Pi Pico W)
import time
import network
import _thread
from umqtt.simple import MQTTClient

# ====== Config ======
# WiFi-instellingen van het lokale netwerk.
WIFI_SSID = "Eureka_wifi"
WIFI_PASSWORD = "EurekaEigenWifi"

# MQTT broker en inloggegevens.
MQTT_BROKER = "10.1.1.121"
MQTT_PORT = 1883
MQTT_USER = "asvzmqtt"
MQTT_PASSWORD = "asvzmqtt"

# Unieke client-id zodat de broker deze Pico W kan herkennen.
CLIENT_ID = "pico_w_01"

# Topic-opbouw: asvz/<woning>/<pomp>/status
WONING = "woning_a"
POMP = "pomp_3"
STATUS_TOPIC = f"asvz/{WONING}/{POMP}/status"

# Na 2 minuten zonder "actief" input schakelt het script terug naar rust.
IDLE_SECONDS = 120  # 2 minuten
# Heartbeat houdt de laatst bekende status actief op de broker.
HEARTBEAT_SECONDS = 5
# ====================

# Gedeelde status tussen de hoofdloop en de input-thread.
last_active_ms = 0
current_status = "rust"
last_publish_ms = 0
lock = _thread.allocate_lock()

def wifi_connect(timeout_s=20):
    print("[wifi] connecting...")
    wlan = network.WLAN(network.STA_IF)
    wlan.active(True)

    if not wlan.isconnected():
        wlan.connect(WIFI_SSID, WIFI_PASSWORD)
        start = time.time()
        while not wlan.isconnected():
            # Stop als verbinden te lang duurt, zodat het script niet blijft hangen.
            if time.time() - start > timeout_s:
                print("[wifi] ERROR: timeout")
                return None
            time.sleep(0.5)

    print("[wifi] connected:", wlan.ifconfig())
    return wlan

def mqtt_connect():
    print("[mqtt] connecting...")
    try:
        client = MQTTClient(
            client_id=CLIENT_ID,
            server=MQTT_BROKER,
            port=MQTT_PORT,
            user=MQTT_USER,
            password=MQTT_PASSWORD,
            keepalive=60
        )
        # Een connectie voor alle statusberichten van deze Pico W.
        client.connect()
        print("[mqtt] connected")
        return client
    except Exception as e:
        print("[mqtt] ERROR:", e)
        return None

def publish_status(client, status):
    try:
        # retain=True zorgt dat nieuwe subscribers direct de laatste status zien.
        client.publish(STATUS_TOPIC, status.encode(), retain=True)
        print(f"[mqtt] status => {status}")
    except Exception as e:
        print("[mqtt] publish ERROR:", e)

def input_thread(client):
    global last_active_ms, current_status
    while True:
        try:
            # Verwacht handmatige input via serial console: "actief" of "rust".
            line = input().strip().lower()
        except Exception as e:
            print("[input] ERROR:", e)
            time.sleep(1)
            continue

        if line in ("actief", "rust"):
            with lock:
                current_status = line
                if line == "actief":
                    # Alleen "actief" ververst de activity-timer.
                    last_active_ms = time.ticks_ms()
            publish_status(client, line)
            print(f"[input] command => {line}")

def main():
    global last_active_ms, current_status, last_publish_ms

    if not wifi_connect():
        return

    client = mqtt_connect()
    if not client:
        return

    _thread.start_new_thread(input_thread, (client,))

    with lock:
        # Start altijd vanuit rust zodat de initiele brokerstatus eenduidig is.
        last_active_ms = time.ticks_ms()
        current_status = "rust"
    publish_status(client, "rust")
    last_publish_ms = time.ticks_ms()

    print("Typ 'actief' om actief te zetten.")
    print("Na 2 minuten zonder input => rust.")

    while True:
        now = time.ticks_ms()

        with lock:
            status = current_status
            idle_ms = time.ticks_diff(now, last_active_ms)

        # Zet status automatisch terug naar rust na te lange inactiviteit.
        if status == "actief" and idle_ms > IDLE_SECONDS * 1000:
            with lock:
                current_status = "rust"
            publish_status(client, "rust")

        # Publiceer periodiek opnieuw zodat monitors weten dat het device nog leeft.
        if time.ticks_diff(now, last_publish_ms) > HEARTBEAT_SECONDS * 1000:
            with lock:
                status = current_status
            publish_status(client, status)
            last_publish_ms = now

        time.sleep(0.2)

if __name__ == "__main__":
    main()
